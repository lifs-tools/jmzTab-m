015public class MZTabErrorType implements Serializable {
016    public enum Category {
017        Format,                // single field format error
018        Logical,               // exists logical error among fields value.
019        CrossCheck             // cross check validation by calling outside files or web services.
020    }
021
022    /**
023     * We classify diagnosis message types into three levels: Info, Warn and Error, all of them coming from
024     * the notational conventions of the specification. All diagnosis rules, which use "MUST", "MUST NOT",
025     * "REQUIRED" keywords, are defined for Error level; use "SHALL", "SHALL NOT", "SHOULD",
026     * "SHOULD NOT" keywords are defined for Warn Level; and others keywords, such as "RECOMMENDED",
027     * "MAY" and "OPTIONAL", are defined for Info Level. 
028     * 
029     */
030    public enum Level {
031        Info,
032        Warn,
033        Error;
034    }
035
036    private Integer code;
037    private Category category;
038    private Level level;
039    private String original;
040    private String cause;
041
042    /**
043     * <p>Constructor for MZTabErrorType.</p>
044     */
045    protected MZTabErrorType() {}
046
047    /**
048     * Code: Unique number for error/warn
049     * Category: Currently, there are three types of messages: Format, Logical, CrossCheck
050     * Original: Message expression pattern. "{?}" is a couple of parameters which can be filled during validate processing.
051     * Cause: A readable text to describe the reason why raise this error/warn. Currently, these cause message coming from mztab specification mainly.
052     */
053    private MZTabErrorType(Integer code, Category category, Level level, String original, String cause) {
054        this.code = code;
055
056        if (category == null) {
057            throw new NullPointerException("MZTabErrorType category can not set null!");
058        }
059        this.category = category;
060
061        this.level = level == null ? Level.Info : level;
062
063        if (original == null || original.trim().length() == 0) {
064            throw new IllegalArgumentException("Original " + original + " is empty!");
065        }
066        this.original = original.trim();
067        this.cause = cause;
068    }
069    
070    public static MZTabErrorType forLevel(MZTabErrorType.Category category, Level level, String keyword) {
071        return MZTabErrorType.createMZTabError(category, level, keyword);
072    }

104        return MZTabErrorType.createMZTabError(category, Level.Info, keyword);
105    }
106
107    /**
108     *  In *_error.properties file, code_{keyword}, original_{keyword}, cause+{keyword} have
109     *  stable format. Thus, this method used to load these properties and create a error.
110     */
111    private static MZTabErrorType createMZTabError(Category category, Level level, String keyword) {
112        if (MZTabStringUtils.isEmpty(keyword)) {
113            throw new NullPointerException(keyword + " can not empty!");
114        }
115
116        String prefix = null;
117        switch (category) {
118            case Format:
119                prefix = "f_";
120                break;
121            case Logical:
122                prefix = "l_";
123                break;
124            case CrossCheck:
125                prefix = "c_";
126                break;
127        }
128
129        Integer code = new Integer(MZTabProperties.getProperty(prefix + "code_" + keyword));
130        String original = MZTabProperties.getProperty(prefix + "original_" + keyword);
131        String cause = MZTabProperties.getProperty(prefix + "cause_" + keyword);
132
133        return new MZTabErrorType(code, category, level, original, cause);
134    }

181    @Override
182    public int hashCode() {
183        int hash = 3;
184        hash = 29 * hash + Objects.hashCode(this.code);
185        hash = 29 * hash + Objects.hashCode(this.category);

191    @Override
192    public boolean equals(Object obj) {
193        if (this == obj) {
194            return true;
195        }
196        if (obj == null) {
197            return false;
198        }
199        if (getClass() != obj.getClass()) {
200            return false;
201        }
202        final MZTabErrorType other = (MZTabErrorType) obj;
203        if (!Objects.equals(this.original, other.original)) {
204            return false;
205        }
206        if (!Objects.equals(this.cause, other.cause)) {
207            return false;
208        }
209        if (!Objects.equals(this.code, other.code)) {

211        }
212        if (this.category != other.category) {
213            return false;
214        }
215        return true;
216    }

218    /**
219     * Code: Unique number for error/warn
220     * Category: Currently, there are three types of messages: Format, Logical
221     * Original: Message expression pattern. "{?}" is a couple of parameters which can be filled during validate processing.
222     * Cause: A readable text to describe the reason why raise this error/warn. Currently, these cause message coming from mztab specification mainly.
223     *
224     * @return a {@link java.lang.String} object.
225     */
226    @Override
227    public String toString() {
228        return  "    Code:\t" + code + "\r\n" +
229                "Category:\t" + category + "\r\n" +
230                "Original:\t" + original + "\r\n" +
231                "   Cause:\t" + (cause == null ? "" : cause) + "\r\n";
232    }

234    /**
235     * <p>findLevel.</p>
236     *
237     * @return the {@link uk.ac.ebi.pride.jmztab2.utils.errors.MZTabErrorType.Level}
238     * @param target a {@link java.lang.String} object.
239     */
240    public static Level findLevel(String target) {
241        if (MZTabStringUtils.isEmpty(target)) {
242            return null;
243        }
244
245        target = MZTabStringUtils.toCapital(target.trim());
246
247        Level level;
248        try {
249            level = Level.valueOf(target);
250        } catch (IllegalArgumentException e) {
251            level = null;
252        }
253
254        return level;
255    }
