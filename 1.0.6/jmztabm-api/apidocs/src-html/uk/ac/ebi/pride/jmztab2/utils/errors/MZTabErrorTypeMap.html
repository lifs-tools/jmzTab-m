001package uk.ac.ebi.pride.jmztab2.utils.errors;
002
003import java.util.Arrays;
004import java.util.Map;
005import java.util.TreeMap;
006
007/**
008 * This list class used to storage all types of MZTabError, and provide query service based on
009 * the error/warn code. This class used to generate mzTab help document.
010 *
011 * @author qingwei
012 * @since 27/02/13
013 * 
014 */
015public final class MZTabErrorTypeMap {
016    private final Map<Integer, MZTabErrorType> typeMap = new TreeMap<Integer, MZTabErrorType>();
017
018    private void add(MZTabErrorType type) {
019        typeMap.put(type.getCode(), type);
020    }
021
022    /**
023     * <p>Constructor for MZTabErrorTypeMap.</p>
024     */
025    public MZTabErrorTypeMap() {
026        Arrays.stream(FormatErrorType.getValues()).forEach((error) -> {
027            add(error);
028        });
029        Arrays.stream(LogicalErrorType.getValues()).forEach((error) -> {
030            add(error);
031        });
032        Arrays.stream(CrossCheckErrorType.getValues()).forEach((error) -> {
033            add(error);
034        });
035    }
036
037    /**
038     * <p>getType.</p>
039     *
040     * @param code a int.
041     * @return a {@link uk.ac.ebi.pride.jmztab2.utils.errors.MZTabErrorType} object.
042     */
043    public MZTabErrorType getType(int code) {
044        return typeMap.get(code);
045    }
046
047    /**
048     * <p>Getter for the field <code>typeMap</code>.</p>
049     *
050     * @return a {@link java.util.Map} object.
051     */
052    public Map<Integer, MZTabErrorType> getTypeMap() {
053        return typeMap;
054    }
055}
