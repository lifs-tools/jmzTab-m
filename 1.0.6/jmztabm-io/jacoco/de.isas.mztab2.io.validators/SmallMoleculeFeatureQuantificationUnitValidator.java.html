package de.isas.mztab2.io.validators;

import de.isas.mztab2.model.Metadata;
import java.util.Arrays;
import java.util.Collections;
import java.util.List;
import uk.ac.ebi.pride.jmztab2.utils.errors.LogicalErrorType;
import uk.ac.ebi.pride.jmztab2.utils.errors.MZTabError;
import uk.ac.ebi.pride.jmztab2.utils.parser.MZTabParserContext;

/**
 * Validates that smallMoleculeFeatureQuantificationUnit is neither null nor
 * empty.
 *
 * @author nilshoffmann
 */
public class SmallMoleculeFeatureQuantificationUnitValidator implements RefiningValidator<Metadata> {

    @Override
    public List<MZTabError> validateRefine(Metadata metadata, MZTabParserContext parserContext) {
        if (metadata.getSmallMoleculeFeatureQuantificationUnit() == null) {
            return Arrays.asList(new MZTabError(
                    LogicalErrorType.NotDefineInMetadata, -1,
                    Metadata.Properties.smallMoleculeFeatureQuantificationUnit + ""));
        }
        return Collections.emptyList();
    }

}
